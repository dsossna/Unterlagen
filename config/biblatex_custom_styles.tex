\newbibmacro*{subtitle+booktitle}{%
  \iffieldsequal{booktitle}{subtitle}{
    \clearfield{subtitle}
  }
  {\iffieldundef{booktitle}
    {}
    {\printfield{subtitle}
     \setunit{\addcolon\space}%
     \printfield{booktitle}
     \setunit{\addcomma}%
     \newunit\newblock
    }
  }%
  \newunit}
  
\DeclareFieldFormat{chapter}{Kapitel~#1}
\DeclareFieldFormat{edition}{Release\addcolon\space~#1\addcomma}
\DeclareFieldFormat{url}{\newline\mkbibacro{URL}\addcolon\space\small{\url{#1}}}

  \usebibmacro{bibindex}%
  \usebibmacro{begentry}%
  \usebibmacro{subtitle+booktitle}%
  \setunit{\labelnamepunct}\newblock
  \usebibmacro{byauthor}%
  \newunit
  \printfield{edition}%
  \newunit\newblock
  \printfield{chapter}%
  \newunit\newblock
  \printfield{url}%
  \usebibmacro{finentry}}
